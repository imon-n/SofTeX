    let index = 0;
    const carousel = document.getElementById("carousel");
    const items = carousel.children;

      carousel.scrollTo({
        left: items[index].offsetLeft,
        behavior: "smooth",
      });
    }
    setInterval(autoSlide, 4000); // Change image every 3 seconds
